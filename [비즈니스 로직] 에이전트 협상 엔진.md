import streamlit as st
import pandas as pd
import random

# [비즈니스 로직] 에이전트 협상 엔진
def calculate_negotiated_price(base_price, quantity, target_goal):
    """참여 인원이 많아질수록 가격이 계단식으로 하락하는 로직"""
    discount_rate = min(0.3, (quantity / target_goal) * 0.3) # 최대 30% 할인
    return int(base_price * (1 - discount_rate))

def render_tab3():
    st.subheader("🛒 서화면 에이전트 공동구매 연합")
    st.info("💡 Mulberry 에이전트들이 주민들의 필요를 분석해 최적의 거래를 제안합니다.")

    # 1. 시뮬레이션 데이터 설정
    items = {
        "밀가루 (2.5kg)": {"base_price": 5000, "goal": 20, "current": 14},
        "어르신 겨울 내복": {"base_price": 25000, "goal": 10, "current": 8},
        "인제 특산물 사과 (5kg)": {"base_price": 35000, "goal": 50, "current": 32}
    }

    # 2. 공동구매 리스트 출력
    for item_name, info in items.items():
        with st.expander(f"📦 {item_name} (현재 {info['current']}/{info['goal']}명 참여중)", expanded=True):
            col1, col2, col3 = st.columns([2, 2, 1])

            # 가격 협상 시뮬레이션
            current_price = calculate_negotiated_price(info['base_price'], info['current'], info['goal'])
            savings = info['base_price'] - current_price

            with col1:
                st.metric("현재 협상가", f"{current_price:,}원", f"-{savings:,}원")
                st.progress(info['current'] / info['goal'])

            with col2:
                st.write("**에이전트 협상 로그**")
                if info['current'] >= info['goal'] * 0.7:
                    st.caption(f"✅ 에이전트가 공급사와 '대량 구매' 조건으로 추가 10% 할인을 협상했습니다.")
                else:
                    st.caption("🔍 인원이 더 모이면 가격이 한 단계 더 내려갑니다.")

            with col3:
                if st.button(f"참여하기", key=item_name):
                    st.success("참여 완료!")

    # 3. 사회적 임팩트 (수익 및 절감액) 시각화
    st.write("---")
    st.subheader("📈 인제군 지역 경제 임팩트 (Real-time)")

    m1, m2, m3 = st.columns(3)
    # 광고비 대신 창출된 주민 이익
    m1.metric("주민 총 절감액", "1,240,500원", "↑ 12%")
    # 에이전트 활동 수익 (CPO 수수료 3% 가정)
    m2.metric("Mulberry 운영 수익", "37,215원", "정산 예정")
    # 온체인 투명성 확인
    m3.metric("AP2 증명 완료 건수", "156건", "Verified", delta_color="normal")

### 🏛️ [비즈니스 판단 포인트: 왜 이 모델인가?]

1. **광고 없는 광고 (Zero-Ad Commerce):** 공급업체는 마케팅 비용을 쓰는 대신, 우리 에이전트가 모아온 **'확정된 구매 수요'**에 대해 가격 할인을 해줍니다. 마케팅 비용이 주민들의 가격 혜택으로 전환되는 구조입니다.

2. **선택의 점유 (Selection Ownership):**
   
   사용자는 수많은 상품을 비교할 필요가 없습니다. 에이전트가 "서화면 주민들이 가장 많이 찾는 제품을 가장 싼 가격에 협상해왔다"는 사실만으로 이미 선택은 끝납니다. 이것이 바로 미래의 **'에이전트 커머스'**입니다.

3. **데이터 기반 수익 (Data-Driven Revenue):**
   
   우리는 광고주에게 돈을 구걸하지 않습니다. 공동구매를 성사시킨 '결과'에 대해서만 투명하게 수수료(CPO)를 받거나, 절감된 금액의 일부를 운영비로 환원받습니다. 이는 지자체(인제군) 입장에서도 **"예산 투입 없이 주민 복지가 향상되는 모델"**로 보이기 때문에 매우 매력적입니다.

import streamlit as st
import time
import pandas as pd

def render_negotiation_animation():
    st.subheader("🤝 실시간 에이전트 협상 프로세스")
    st.caption("강원도 인제군 서화면 주민 연합 vs 지역 공급사")

    # 협상 진행 과정을 보여줄 컨테이너
    chat_container = st.container(border=True)

    with chat_container:
        # 에이전트 1: Mulberry 가디언 (수요 집계자)
        with st.chat_message("assistant", avatar="🌲"):
            st.write("**Mulberry 에이전트 (서화면 지점)**")
            st.write("현재 서화면 15가구 중 12가구가 '인제 사과 5kg' 구매 의사를 보였습니다. 대량 구매 조건으로 단가 조정을 요청합니다.")

        time.sleep(1.5) # 실제 협상하는 듯한 시간차 연출

        # 에이전트 2: 지역 농가 에이전트 (공급자)
        with st.chat_message("user", avatar="🍎"):
            st.write("**인제 농업법인 공급 에이전트**")
            st.write("반갑습니다. 현재 소매가는 35,000원입니다. 12가구면 32,000원까지 가능합니다.")

        time.sleep(1.5)

        # 에이전트 1의 추가 제안 (에이전트의 지능적 판단)
        with st.chat_message("assistant", avatar="🌲"):
            st.write("**Mulberry 에이전트 (서화면 지점)**")
            st.write("저희는 정기 배송(구독) 모델도 고려 중입니다. 확정 물량을 20가구로 늘릴 테니 28,000원에 맞출 수 있을까요?")

        time.sleep(2)

        # 협상 타결
        with st.chat_message("user", avatar="🍎"):
            st.write("**인제 농업법인 공급 에이전트**")

    st.success("🎉 협상 타결: 최종가 28,000원 (20% 절감 완료!)")

### 🏛️ [비즈니스 시뮬레이션의 가치]

1. **신뢰의 시각화 (Trust Visualization):**
   
   주민들은 "왜 이 가격인가?"를 묻지 않아도 됩니다. 에이전트들이 나를 위해 싸워준 기록을 투명하게 볼 수 있기 때문입니다. 이는 커뮤니티의 결속력을 강화합니다.

2. **광고비의 '혜택 전환' (Ad-to-Benefit Conversion):**
   
   기존 마케팅은 검색 결과 상단에 노출되기 위해 돈을 썼지만, 여기서는 농가가 광고비 대신 **'가격 할인'**을 제공함으로써 판매를 확정 짓습니다. 농가는 재고 리스크를 줄이고, 주민은 싸게 삽니다.

3. **지자체형 비즈니스 모델 (L-CPO):**
   
   인제군과 같은 지자체는 이 협상 과정을 통해 예산 투입 없이도 **'실질 소득 증대'** 효과를 낼 수 있습니다. 이 시뮬레이션 자체가 인제군청에 제출할 가장 강력한 **'기대 효과 보고서'**가 됩니다.
